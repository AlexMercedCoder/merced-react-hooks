## Install

```bash
npm install --save merced-react-hooks
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'merced-react-hooks'
import 'merced-react-hooks/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
